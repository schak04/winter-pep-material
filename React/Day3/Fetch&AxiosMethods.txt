 function App() {
      const [movies, setMovies] = React.useState([]);
      const [movieName, setMovieName] = React.useState("");

      async function fetchMovie() {
        try {
          let response = await fetch(
            "https://reactmovie-7dcf7-default-rtdb.asia-southeast1.firebasedatabase.app/movies.json",
          );
          const result = await response.json();
          const movieArray = Object.entries(result).map(([id, movies]) => ({
            id,
            movies,
          }));
          setMovies(movieArray);
        } catch (error) {
          console.log("Error while fetching data");
        }
      }

      async function addMovie() {
        if (!movieName) {
          alert("Please add the movie name first...");
        }
        try {
          const response = await fetch(
            "https://reactmovie-7dcf7-default-rtdb.asia-southeast1.firebasedatabase.app/movies.json",
            {
              method: "POST",
              body: JSON.stringify({ name: movieName }),
              headers: { "content-type": "application/json" },
            },
          );
        } catch (error) {
          console.log(error);
        }
      }
      return (
        <>
          <h1>Movies</h1>
          <input
            type="text"
            value={movieName}
            placeholder="Enter the movie name"
            onChange={(e) => setMovieName(e.target.value)}
          />
          <button onClick={addMovie}>Add movie</button>
          <button onClick={fetchMovie}>Fetch Movies</button>
          <ul>
            {movies.map((movie, ind) => (
              <li key={ind}>{movie.movies.name}</li>
            ))}
          </ul>
        </>
      );
    }
    const root = ReactDOM.createRoot(document.getElementById("root"));
    root.render(<App />);
